Jquery Form - Custom Rules



Usage

			
				$('#amsify-validate').amsifyForm({
					fieldRules : [
						{
							field	: 'username',
							rules 	: {
								required 	: {},
								safiRule 	: function(value) {
									if($.trim(value) != 'safi') {
										return 'Username should be safi';
									}
									return true;
								},
								lengthRule 	: function(value) {
									if(value.length > 4) {
										return 'length is exceeding';
									}
									return true;
								},
							}
						},
						{
							field	: 'name',
							rules 	: {
								required 	: {},
								safiRule 	: function(value) {
									if($.trim(value) != 'zaid') {
										return 'Username should be zaid';
									}
									return true;
								},
								lengthRule 	: function(value) {
									if(value.length > 4) {
										return 'length is exceeding';
									}
									return true;
								},
							}
						}
					]
				});
