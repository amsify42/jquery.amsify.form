Usage

						fieldRules : [
							{
								field 	: 'required',

									required 	: {}
								}

							{

								rules 	: {
									requiredif 	: {},
									email 		: {}
								}
							},
							{
								field 	: 'email',
								rules 	: {
									required 	: {},
									email 		: {}
								}
							},
							{
								field 	: 'onlynumber',
								rules 	: {
									required 	: {},
									onlynumber 	: {}
								}
							},
							{
								field 	: 'onlydecimal',
								rules 	: {
									required 	: {},
									onlydecimal : {}
								}
							},
							{
								field 	: 'nospecialchar',
								rules 	: {
									required 		: {},
									nospecialchar 	: {}
								}
							},
							{
								field 	: 'alphanumeric',
								rules 	: {
									required 		: {},
									alphanumeric 	: {}
								}
							},
							{
								field 	: 'pattern',
								rules 	: {
									required 	: {},
									pattern 	: {
										check: /^[a-zA-Z0-9- ]*$/
