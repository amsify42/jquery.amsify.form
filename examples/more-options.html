Jquery Form - AutoValidate and ValidateOn



AutoValidate: false
secureAttributes: true

Usage

			
				$('.amsify-validate2').amsifyForm({
					autoValidate: false,
					secureAttributes: true
				});


ValidateOn : keyup focusout
loadingText: 'Loading...'

Usage

				
					$('.amsify-validate3').amsifyForm({
						validateOn: 'keyup focusout',
						loadingText: 'Loading...'
					});


submitSelector: '#my-submit'
errorClass : '.custom-error'

Usage

				
					$('.amsify-validate4').amsifyForm({
						submitSelector: '#my-submit',
						errorClass : '.custom-error'
					});
