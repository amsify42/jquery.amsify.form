Jquery Form - Ajax



Usage

			
				AmsifyHelper.base_url += '/jquery-plugins/form';
				$('#amsify-validate').amsifyForm({
					submit 		:'#amsify-submit',
					loadingText	:'Authorising...',
					ajax 		: {
						action 		: '/test.php',
						callback 	: function(data) {
							alert('Ajax Responded');
							console.info(data);
						}
					}
				});
