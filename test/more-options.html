Jquery Form - AutoValidate and ValidateOn



AutoValidate: false



ValidateOn : focusout



Error Class : focusout

Usage

			
				AmsifyHelper.base_url += '/jquery-plugins/form';
				$('.amsify-validate2').amsifyForm({
					autoValidate: false
				});
				$('.amsify-validate3').amsifyForm({
					validateOn: 'focusout'
				});
